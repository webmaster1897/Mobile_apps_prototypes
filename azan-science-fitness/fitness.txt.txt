<script>
function switchTab(viewName){
  document.getElementById('view-profile').classList.remove('active');
  document.getElementById('view-feed').classList.remove('active');
  document.getElementById('nav-profile').classList.remove('active');
  document.getElementById('nav-home').classList.remove('active');

  if(viewName === 'profile'){
    document.getElementById('view-profile').classList.add('active');
    document.getElementById('nav-profile').classList.add('active');
  } else {
    document.getElementById('view-feed').classList.add('active');
    document.getElementById('nav-home').classList.add('active');
  }
}

function toggleLike(heartId, countId) {
  var heart = document.getElementById(heartId);
  var countSpan = document.getElementById(countId);
  var currentLikes = parseInt(countSpan.innerText);
  
  if(heart.classList.contains('active')) {
    heart.classList.remove('active');
    countSpan.innerText = currentLikes - 1;
  } else {
    heart.classList.add('active');
    heart.innerText = "❤️";
    countSpan.innerText = currentLikes + 1;
  }
}

function setReaction(id, emoji) {
  var heart = document.getElementById(id);
  heart.classList.add('active');
  heart.innerText = emoji;
}
</script>
